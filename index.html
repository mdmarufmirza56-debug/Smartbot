async function sendMessage() { const input = document.getElementById('message-input'); const message = input.value.trim(); const chat = getCurrentChat(); if (!message || isGenerating || !state.settings.apiKey) { if (!state.settings.apiKey) openModal('settings-modal'); return; } // 1. Update State & UI const userMsg = { role: 'user', content: message, id: Date.now() }; chat.messages.push(userMsg); renderMessages(); input.value = ''; autoResize(input); isGenerating = true; // 2. Prepare API Call try { const response = await fetch("https://openrouter.ai/api/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${state.settings.apiKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: state.settings.model || "google/gemini-2.0-flash-exp:free", messages: [ { role: "system", content: state.settings.systemPrompt }, ...chat.messages.map(m => ({ role: m.role, content: m.content })) ] }) }); const data = await response.json(); const aiMessage = data.choices[0].message.content; // 3. Store AI Response chat.messages.push({ role: 'assistant', content: aiMessage, id: Date.now() + 1 }); renderMessages(); } catch (error) { console.error("Chat Error:", error); } finally { isGenerating = false; saveState(); } }
